    // === ２次元配列作成   
    for($i = 0; $i < $h; $i++) {
        $arr[] = explode(" ", fgets(STDIN));
    }
    
  // === ２次元配列の　行　の　和を求める
   $arr_sum = [];    
   for($i = 0; $i < $h; $i++) {
       for($j = 0; $j < $w; $j++) {
            $arr_sum[$i] += $arr[$i][$j];

   }

    // Let's チャレンジ！！
   fscanf(STDIN, "%d %d", $h, $w);
   
   for($i = 0; $i < $h; $i++) {
       $arr[] = explode(' ', trim(fgets(STDIN)));
   }
   
   $ans = $arr[0][0];
   
  // === 最大値を入れる　全探索  配列の１番目の要素と比較して、それより大きい場合は ===
  // 大きい要素を $ans に入れる。
   for($i = 0; $i < $h; $i++) {
       for($j = 0; $j < $w; $j++) {
           if($ans < $arr[$i][$j]) {
               $ans = $arr[$i][$j];
           }
       }
   }
   print($ans);
